function write(text) {
    const out = document.getElementById("output");
    out.innerText += "\n" + text;
    out.scrollTop = out.scrollHeight;
}

function boot() {
    write("INITIALIZING NOCTYRA ARCHIVE TERMINAL...");
    write("LINKING OBSERVATION NODE...");
    write("STATUS: LIMITED ACCESS GRANTED");
    write("\nCOMMANDS:");
    write("- LIST");
    write("- OPEN <file_id>");
    write("- CLEAR");
    write("- HELP");
}

function listFiles() {
    write("\nREGISTERED FILES:");
    for (let key in files) {
        write(" > " + key);
    }
}

function openFile(id) {
    const sound = document.getElementById("pageSound");
    sound.currentTime = 0;
    sound.play();

    if (!files[id]) {
        write("\nACCESS DENIED // FILE NOT FOUND");
        return;
    }

    write("\nOPENING FILE: " + id);
    write("--------------------------------");
    write(files[id]);
    write("--------------------------------");
}

function clearScreen() {
    document.getElementById("output").innerText = "";
}

function help() {
    write("\nCOMMAND REFERENCE:");
    write("LIST            - show archive index");
    write("OPEN <id>       - access file");
    write("CLEAR           - reset terminal view");
}

document.getElementById("cmd").addEventListener("keydown", function(e) {
    if (e.key === "Enter") {
        const input = this.value.trim();
        this.value = "";

        if (!input) return;

        write("\n> " + input);

        const args = input.split(" ");
        const cmd = args[0].toLowerCase();

        if (cmd === "list") listFiles();
        else if (cmd === "open") openFile(args[1]);
        else if (cmd === "clear") clearScreen();
        else if (cmd === "help") help();
        else write("UNKNOWN COMMAND");
    }
});

boot();
